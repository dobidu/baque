<objective>
## Goal
Add per-step timing offset to the sequencer pipeline: `FeelPattern` holds `timing_ms` and `vel_scale`
per step; `FeelEngine` manages a pre-allocated deferred note queue for notes that offset beyond the
current block boundary. Sequencer::generate() flushes deferred notes then applies offsets during
note generation. PluginProcessor owns and wires the new types.

## Purpose
First slice of Phase 5 (Feel Engine — product core). Establishes the offset-and-defer mechanism
that every feel effect builds on. Phase 05-02 adds Gaussian humanize + PRNG seeding on top of this
infrastructure; 05-03 adds feel presets (Dilla Drunk, Burial Broken, etc.).

- `src/audio/feel_pattern.h` — FeelStep + FeelPattern POD structs (16 steps, timing_ms + vel_scale)
- `src/audio/feel_engine.h/.cpp` — FeelEngine: deferred note queue, flush, defer, offset calc
- `src/audio/sequencer.h/.cpp` — generate() extended with nullable feel params (backward-compat)
- `src/plugin_processor.h/.cpp` — FeelEngine + FeelPattern members; block_start_sample_ tracking
- `tests/test_feel_engine.cpp` — 8 deterministic tests (FE1..FE8)

### 1.2 Create `src/audio/feel_engine.h`

```cpp
#pragma once
#include <juce_audio_basics/juce_audio_basics.h>
#include <cstdint>

// Manages deferred MIDI notes (offsets that push past block boundary) and per-step timing math.
// NOT RT-allocated: deferred_ is a fixed array. All methods noexcept.
class FeelEngine {
  public:
    static constexpr int k_max_deferred = 128;

    struct DeferredNote {
        juce::MidiMessage msg{};
        int64_t target_sample = 0;
    };

    // Clear deferred queue. Call from prepareToPlay() and on transport start.
    void prepare() noexcept;

    // Convert timing_ms to sample count delta. Pure, no state.
    [[nodiscard]] static int timing_ms_to_samples(float timing_ms, double sample_rate) noexcept;

    // Flush deferred notes whose target falls within [block_start, block_start + block_size).
    // Adds matching notes to buf. Remaining notes stay in queue.
    void flush_deferred(juce::MidiBuffer& buf, int64_t block_start, int block_size) noexcept;

    // Add a note to the deferred queue. Drops + jassertfalse if queue full.
    void defer(const juce::MidiMessage& msg, int64_t target_sample) noexcept;

    [[nodiscard]] int deferred_count() const noexcept { return deferred_count_; }

  private:
    DeferredNote deferred_[k_max_deferred]{};
    int deferred_count_ = 0;
};
```

### 1.3 Create `src/audio/feel_engine.cpp`

```cpp
#include "feel_engine.h"
#include <algorithm>
#include <cmath>

void FeelEngine::prepare() noexcept {
    deferred_count_ = 0;
}

int FeelEngine::timing_ms_to_samples(float timing_ms, double sample_rate) noexcept {
    return static_cast<int>(std::round(static_cast<double>(timing_ms) * sample_rate / 1000.0));
}

void FeelEngine::flush_deferred(juce::MidiBuffer& buf, int64_t block_start, int block_size) noexcept {
    const int64_t block_end = block_start + static_cast<int64_t>(block_size);
    int write = 0;
    for (int i = 0; i < deferred_count_; ++i) {
        const auto& d = deferred_[i];
        if (d.target_sample >= block_start && d.target_sample < block_end) {
            buf.addEvent(d.msg, static_cast<int>(d.target_sample - block_start));
        } else if (d.target_sample >= block_end) {
            deferred_[write++] = deferred_[i];  // future — keep
        } else {
            jassertfalse;  // audit-added (M2): stale past note — discard; queue leak prevention
        }
    }
    deferred_count_ = write;
}

void FeelEngine::defer(const juce::MidiMessage& msg, int64_t target_sample) noexcept {
    jassert(msg.getRawDataSize() <= 3);  // audit-added (M4): RT-safety — only inline MIDI msgs (note-on/off ≤3 bytes); larger msgs allocate on copy
    jassert(deferred_count_ < k_max_deferred);
    if (deferred_count_ >= k_max_deferred) return;  // RT-safe drop
    deferred_[deferred_count_++] = {msg, target_sample};
}
```

Tests use a helper `make_transport(ppq, bpm)` and simulate 1–2 block calls to verify feel behavior.

```cpp
#include <catch2/catch_test_macros.hpp>
#include <catch2/catch_approx.hpp>
#include <juce_audio_basics/juce_audio_basics.h>
#include "audio/feel_engine.h"
#include "audio/feel_pattern.h"
#include "audio/sequencer.h"
#include "audio/step_pattern.h"
#include "audio/transport_state.h"

struct FeelJuceFixture {
    juce::ScopedJuceInitialiser_GUI init;
};

static TransportState make_transport(double ppq = 0.0, double bpm = 120.0) {
    TransportState t{};
    t.is_playing    = true;
    t.ppq_position  = ppq;
    t.bpm           = bpm;
    return t;
}

// Build a StepPattern where lane 0 fires on step 0 only.
static StepPattern single_step_pattern() {
    StepPattern p{};
    p.set_active(0, 0, true);
    p.set_note(0, 0, 36);
    return p;
}

// Build a StepPattern where all lanes fire on all steps.
static StepPattern all_steps_pattern() {
    StepPattern p{};
    for (int s = 0; s < StepPattern::k_num_steps; ++s) {
        p.set_active(0, s, true);
        p.set_note(0, s, 36);
    }
    return p;
}

TEST_CASE("FE1 - positive timing offset shifts note forward", "[feel]") {
    FeelJuceFixture f;
    const double sr = 44100.0;
    const int block = 8192;

    FeelPattern feel{};
    feel.enabled = true;
    feel.steps[0].timing_ms = 25.0f;

    FeelEngine engine;

    Sequencer seq;
    seq.set_pattern(single_step_pattern());  // audit-added (M1): set_next_pattern requires 15→0 boundary; use set_pattern for immediate effect

    juce::MidiBuffer buf;
    seq.generate(make_transport(0.0), buf, block, sr, &feel, &engine, 0);

    // Find note-on for note 36 in buf
    int note_on_pos = -1;
    for (const auto meta : buf) {
        if (meta.getMessage().isNoteOn() && meta.getMessage().getNoteNumber() == 36)
            note_on_pos = meta.samplePosition;
    }
    REQUIRE(note_on_pos >= 0);
    // 25ms at 44100 = 1102.5 → 1103 (rounded)
    const int expected = static_cast<int>(std::round(25.0 * sr / 1000.0));
    REQUIRE(note_on_pos >= expected - 2);
    REQUIRE(note_on_pos <= expected + 2);
}

    FeelJuceFixture f;
    const double sr = 44100.0;
    const int block = 4096;

    FeelPattern feel{};

    feel.steps[0].timing_ms = 150.0f;  // 6615 samples at 44100 — beyond 4096

TEST_CASE("FE7 - all 16 steps with distinct offsets fire at correct positions", "[feel]") {
    FeelJuceFixture f;
    const double sr = 44100.0;
    // At 120 BPM: 1 pattern = 4 beats = 16 steps. 1 beat = 44100/2 = 22050 samples.
    // 1 step = 22050/4 = 5512.5 samples. Full pattern = 88200 samples; use 98304 (safe).
    const int block = 98304;
    const double bpm = 120.0;

    FeelPattern feel{};
    feel.enabled = true;
    for (int i = 0; i < 16; ++i)
        feel.steps[i].timing_ms = static_cast<float>((i - 8) * 3);  // −24ms to +21ms

    FeelEngine engine;
    Sequencer seq;
    seq.set_pattern(all_steps_pattern());  // audit-added (M1)

    juce::MidiBuffer buf;
    seq.generate(make_transport(0.0, bpm), buf, block, sr, &feel, &engine, 0);

    // Collect note-on positions (note 36, lane 0)
    std::vector<int> actual_positions;
    for (const auto meta : buf)
        if (meta.getMessage().isNoteOn() && meta.getMessage().getNoteNumber() == 36)
            actual_positions.push_back(meta.samplePosition);

    // audit-added (SR1): must have exactly 16 note-ons (one per step); count-only check was insufficient
    REQUIRE(actual_positions.size() == 16);

    // audit-added (SR1): verify each note fires at expected position ± 3 samples
    // step_samples = sr * 60 / bpm / 4 (quarter-note / 4 = 16th note)
    const double step_samples = sr * 60.0 / bpm / 4.0;  // = 5512.5 at 120BPM/44100Hz
    for (int i = 0; i < 16; ++i) {
        const double base = std::round(i * step_samples);
        const int offset = FeelEngine::timing_ms_to_samples(feel.steps[i].timing_ms, sr);
        const int expected = static_cast<int>(base) + offset;
        const int clamped_expected = std::max(0, expected);
        REQUIRE(actual_positions[i] >= clamped_expected - 3);
        REQUIRE(actual_positions[i] <= clamped_expected + 3);
    }
}

TEST_CASE("FE8 - deferred queue overflow drops note and triggers jassertfalse", "[feel]") {
    FeelJuceFixture f;

    juce::MidiMessage msg = juce::MidiMessage::noteOn(1, 60, static_cast<uint8_t>(100));

    for (int i = 0; i < FeelEngine::k_max_deferred; ++i)
        engine.defer(msg, static_cast<int64_t>(i) * 100);

    // Use JUCE_ASSERT_MESSAGE_BOX_OFF or test in release; here just verify no allocation/exception
    // and count does not exceed cap
    engine.defer(msg, 999999);
    REQUIRE(engine.deferred_count() == FeelEngine::k_max_deferred);
}
```
